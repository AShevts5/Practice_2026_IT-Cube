# 1) Перейти в проект
cd Frontend
# 2) Установить зависимости
npm install
# 3) Запуск dev-сервера
npm run dev
Полезные команды
# Линт
npm run lint
# Сборка production
npm run build
# Проверка (lint + build)
npm run check
# Проверка + запуск dev
npm run dev:check
# Локальный просмотр production-сборки
npm run preview
# Перегенерация API-типов из OpenAPI
npm run api
